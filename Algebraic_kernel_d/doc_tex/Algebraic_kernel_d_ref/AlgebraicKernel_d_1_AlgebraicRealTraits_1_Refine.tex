\ccMethod{
result_type operator()(const argument_type & x);}{ 
  This operator at least half's the current interval of $x$. \\
  Note that an interval may also be degenerated to a single point.
} 

\ccMethod{
result_type operator()(const argument_type & x, int rel_prec);}{ 
  This operator refines the current interval of $x$ with respect to the given relative precision. \\
  That is: $|lower - upper| / |x| \leq 2^{-rel\_prec}$\\
  In case $x$ equals zero the interval degenerates to the zero interval. 
}
