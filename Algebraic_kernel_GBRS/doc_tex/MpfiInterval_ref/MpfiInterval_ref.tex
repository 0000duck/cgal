\begin{ccRefClass}{MpfiInterval}

\ccIsModel{Algebraic\_real\_1}

\ccDefinition

An object of the class \ccStyle{MpfiInterval} is an interval, whose endpoints
are arbitrary precision floating point numbers (implemented using the
{\sc MPFR} library).

\ccInclude{CGAL/MpfiInterval.h}

\ccCreationVariable{i}
\ccCreation

\ccConstructor{MpfiInterval();}
	{creates an interval object without setting its endpoints.}

\ccConstructor{MpfiInterval (int);}{} \ccGlue
\ccConstructor{MpfiInterval (unsigned int);}{} \ccGlue
\ccConstructor{MpfiInterval (long int);}{} \ccGlue
\ccConstructor{MpfiInterval (unsigned long int);}{} \ccGlue
\ccConstructor{MpfiInterval (double);}{} \ccGlue
\ccConstructor{MpfiInterval (mpz_t);}{} \ccGlue
\ccConstructor{MpfiInterval (mpq_t);}{} \ccGlue
\ccConstructor{MpfiInterval (CGAL::Gmpq);}{} \ccGlue
\ccConstructor{MpfiInterval (CGAL::Gmpz);}{} \ccGlue
\ccConstructor{MpfiInterval (mpfr_t);}
	{creates an interval object whose endpoints are the same number.}

\ccConstructor{MpfiInterval (int, int);}{} \ccGlue
\ccConstructor{MpfiInterval (unsigned int, unsigned int);}{} \ccGlue
\ccConstructor{MpfiInterval (long int, long int);}{} \ccGlue
\ccConstructor{MpfiInterval (unsigned long int, unsigned long int);}{} \ccGlue
\ccConstructor{MpfiInterval (double, double);}{} \ccGlue
\ccConstructor{MpfiInterval (mpz_t, mpz_t);}{} \ccGlue
\ccConstructor{MpfiInterval (mpq_t, mpq_t);}{} \ccGlue
\ccConstructor{MpfiInterval (CGAL::Gmpq, CGAL::Gmpq);}{} \ccGlue
\ccConstructor{MpfiInterval (CGAL::Gmpz, CGAL::Gmpz);}{} \ccGlue
\ccConstructor{MpfiInterval (mpfr_t, mpfr_t);}
	{creates an interval object given its two endpoints.}

\ccConstructor{MpfiInterval (mpfi_t);}{} \ccGlue
\ccConstructor{MpfiInterval (MpfiInterval);}
	{creates an interval object given another interval}

\ccOperations
\ccSetThreeColumns{MpfiInterval}{}{\hspace*{8cm}}

\ccMethod{mpfi_t mpfi();}{returns the MPFI value stored.}
\ccMethod{void set_prec(mp_prec_t p);}{sets the precision of the interval to $p$.}
\ccMethod{mp_prec_t get_prec ();}{gets the current precision of the interval.}
\ccMethod{void get_left (mpfr_t l);}
	{stores in $l$ the value of the left endpoint of the interval.}
\ccMethod{void get_right (mpfr_t r);}
	{stores in $r$ the value of the right endpoint of the interval.}
\ccMethod{void get_endpoints (mpfr_t l, mpfr_t r);}
	{stores in $l$ and $r$ the values of the endpoints of the interval.}
\ccMethod{bool is_point ();}
	{returns true iff the two endpoints of the interval are equal.}

\ccMethod{bool contains (int n);}{} \ccGlue
\ccMethod{bool contains (mpfr_t n);}{} \ccGlue
\ccMethod{bool contains (mpz_t n);}{} \ccGlue
\ccMethod{bool contains (mpq_t n);}{} \ccGlue
\ccMethod{bool contains (Gmpz n);}{} \ccGlue
\ccMethod{bool contains (Gmpq n);}
	{returns true iff the interval contains $n$.}

\ccMethod{bool operator== (n);}{} \ccGlue
\ccMethod{bool operator!= (n);}{} \ccGlue
\ccMethod{bool operator< (n);}{} \ccGlue
\ccMethod{bool operator> (n);}{} \ccGlue
\ccMethod{bool operator<= (n);}{} \ccGlue
\ccMethod{bool operator>= (n);}
	{comparison operators, where $n$ has type int, mpz\_t, mpq\_t, Gmpz, Gmpq, mpfr\_t or MpfiInterval.}

\ccMethod{MpfiInterval operator+ (n);}{} \ccGlue
\ccMethod{MpfiInterval operator- (n);}{} \ccGlue
\ccMethod{MpfiInterval operator* (n);}{} \ccGlue
\ccMethod{MpfiInterval operator/ (n);}{} \ccGlue
\ccMethod{MpfiInterval operator+= (n);}{} \ccGlue
\ccMethod{MpfiInterval operator-= (n);}{} \ccGlue
\ccMethod{MpfiInterval operator*= (n);}{} \ccGlue
\ccMethod{MpfiInterval operator/= (n);}
	{arithmetic operators, where $n$ has type int, mpz\_t, mpq\_t, Gmpz, Gmpq, mpfr\_t or MpfiInterval.}

\end{ccRefClass}
