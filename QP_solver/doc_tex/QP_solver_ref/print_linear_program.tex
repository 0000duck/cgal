\begin{ccRefFunction}{print_linear_program}

\ccInclude{CGAL/QP_functions.h}

This function writes a linear program to an output stream.

\ccFunction{template <LinearProgramInterface>
void print_linear_program 
(std::ostream& out, const LinearProgramInterface &lp, 
 const std::string& problem_name = std::string("MY_MPS"));}
{writes the linear program \ccc{lp} to \ccc{out} in 
\ccc{MPSFormat}. The name of the program will be the one provided 
by \ccc{problem_name}.}

\ccHeading{Requirements}
Output operators are defined for all entry types of \ccc{lp}.

\end{ccRefFunction}
