\begin{ccRefConcept}{Kinetic::RegularTriangulationVisitor_3}

\ccDefinition

This concept is for proxy objects which get notified when a kinetic regular triangulation changes. It inherits all the methods of \ccc{DelaunayTriangulationVisitor_3}.

\ccIsModel

\ccc{Kinetic::DelaunayTriangulationVisitor_3}


\ccCreationVariable{v}

\ccOperations

\ccMethod{void pre_move(Key, Cell);}{The point defined by \ccc{Key} is about to move from the cell.}

\ccMethod{void post_move(Key, Cell);}{The point defined by \ccc{Key} just moved to the cell.}

\ccHasModels 

\ccc{Kinetic::Regular_triangulation_visitor_base_3}, 
\ccc{Kinetic::Regular_triangulation_event_log_visitor_3}

\end{ccRefConcept}
