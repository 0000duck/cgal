\begin{ccRefConcept}{ArrangementDcelWithRebind}

\ccDefinition
% ===========

The concept \ccRefName{} refines the basic \dcel{} concept by adding
a rebind struct-template.

\ccRefines
\ccc{ArrangementDcel}

\ccTypes
% ======
\ccNestedType{template <class T> rebind}
   {allows the rebinding of any model of this concept with a different
     geometry traits class \ccc{T}.}

\ccCreation
\ccCreationVariable{dcel}
% =======================  
\ccConstructor{Arr_dcel();}
  {constructs an empty \dcel{} with one unbouned face.}

% ==========
