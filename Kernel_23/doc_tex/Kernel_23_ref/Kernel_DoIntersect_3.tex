\begin{ccRefFunctionObjectConcept}{Kernel::DoIntersect_3}
A model for this must provide

\ccCreationVariable{fo}

\ccMemberFunction{bool operator()(Type1 obj1, Type2 obj2);}
{determines if two geometrical objects of type 
\ccStyle{Type1} and \ccStyle{Type2} intersect or not}

for all pairs \ccStyle{Type1} and \ccStyle{Type2}, where
the types \ccStyle{Type1} and 
\ccStyle{Type2} can be any of the following:
\begin{itemize}\ccTexHtml{\itemsep0pt\topsep0pt\partopsep0pt\parskip0pt\parsep0pt}{}
\item \ccStyle{Kernel::Plane_3}
\item \ccStyle{Kernel::Line_3}
\item \ccStyle{Kernel::Ray_3}
\item \ccStyle{Kernel::Segment_3}
\item \ccStyle{Kernel::Triangle_3}
\item \ccStyle{Bbox_3}
\end{itemize}
and also for \ccc{Type1} and \ccc{Type2} of respective types
\begin{itemize}\ccTexHtml{\itemsep0pt\topsep0pt\partopsep0pt\parskip0pt\parsep0pt}{}
\item \ccStyle{Kernel::Triangle_3} and \ccc{Kernel::Tetrahedron_3}
\item \ccStyle{Kernel::Plane_3} and \ccStyle{Kernel::Sphere_3} (or the contrary)
\item \ccStyle{Kernel::Sphere_3} and \ccStyle{Kernel::Sphere_3}.
\end{itemize}

\ccRefines
\ccc{AdaptableFunctor} (with two arguments)
