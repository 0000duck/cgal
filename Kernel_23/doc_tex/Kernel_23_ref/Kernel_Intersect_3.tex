\ccCreationVariable{fo}

\ccMemberFunction{Kernel::Object_3 operator()(Type1 obj1, Type2 obj2);}
{computes the \ccHtmlNoLinksFrom{intersection} region of two geometrical 
objects of type \ccStyle{Type1} and \ccStyle{Type2}}

for all pairs \ccStyle{Type1} and \ccStyle{Type2}, where
the type \ccStyle{Type1} is \ccStyle{Kernel::Plane_3} and 
\ccStyle{Type2} can be any of the following:
\begin{itemize}\ccTexHtml{\itemsep0pt\topsep0pt\partopsep0pt\parskip0pt\parsep0pt}{}
\item \ccStyle{Kernel::Plane_3}
\item \ccStyle{Kernel::Line_3}
\item \ccStyle{Kernel::Ray_3}
\item \ccStyle{Kernel::Segment_3}
\item \ccStyle{Kernel::Triangle_3}
\item \ccStyle{Kernel::Sphere_3}
\end{itemize}
and also where the type \ccc{Type1} is \ccc{Kernel::Triangle_3} and \ccc{Type2} can be any of the following:
\begin{itemize}\ccTexHtml{\itemsep0pt\topsep0pt\partopsep0pt\parskip0pt\parsep0pt}{}
\item \ccStyle{Kernel::Line_3}
\item \ccStyle{Kernel::Ray_3}
\item \ccStyle{Kernel::Plane_3}
\item \ccStyle{Kernel::Segment_3}.
\end{itemize}

\ccMemberFunction{Kernel::Object_3 operator()(Kernel::Line_3 l1,
                                              Kernel::Line_3 l2);}
{computes the \ccHtmlNoLinksFrom{intersection} of two lines.  The result
can be either a \ccc{Kernel::Point_3}, a \ccc{Kernel::Line_3}, or empty.}

\ccMemberFunction{Kernel::Object_3 operator()(Kernel::Segment_3 s,

{computes the \ccHtmlNoLinksFrom{intersection} of a segment and a line.  The result
can be either a \ccc{Kernel::Point_3}, a \ccc{Kernel::Segment_3} or empty.}

\ccMemberFunction{Kernel::Object_3 operator()(Kernel::Segment_3 s1,
                                              Kernel::Segment_3 s2);}
{computes the \ccHtmlNoLinksFrom{intersection} of two segments.  The result
can be either a \ccc{Kernel::Point_3}, a \ccc{Kernel::Segment_3} or empty.}

\ccMemberFunction{Kernel::Object_3 operator()(Kernel::Sphere_3 s1,
                                              Kernel::Sphere_3 s2);}
{computes the \ccHtmlNoLinksFrom{intersection} of two spheres.  The result
can be either a \ccc{Kernel::Point_3}, a \ccc{Kernel::Circle_3}, a 
\ccc{Kernel::Sphere_3} or empty.}

\ccMemberFunction{Kernel::Object_3 operator()(Kernel::Triangle_3 t1,
                                              Kernel::Triangle_3 t2);}
{computes the \ccHtmlNoLinksFrom{intersection} of two triangles.  The result
can be either a \ccc{Kernel::Point_3}, a \ccc{Kernel::Segment_3}, a 
\ccc{Kernel::Triangle_3}, \ccc{std::vector<Kernel::Point_3>} or empty.}

\ccMemberFunction{Kernel::Object_3 operator()(Kernel::Plane_3 pl1,
                                              Kernel::Plane_3 pl2,
                                              Kernel::Plane_3 pl3);}
{computes the \ccHtmlNoLinksFrom{intersection} of three planes.  The result
can be either a \ccc{Kernel::Point_3}, a \ccc{Kernel::Line_3}, a
\ccc{Kernel::Plane_3}, or empty.}
