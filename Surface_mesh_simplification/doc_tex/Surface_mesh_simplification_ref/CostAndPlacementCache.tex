\begin{ccRefConcept}{CostAndPlacementCache}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries

\ccDefinition

The concept \ccRefName\ describes the requirements for the data structure used to cache both the {\em cost} and {\em placement} for an edge.

The cost and placement are stored as \ccc{boost::optional} values (i.e. they can be absent) because the cost and/or placement functions might be unable to compute them (or intentionally wants to prevent a certain edge to be collapsed).

\ccRefines
\ccc{DefaultConstructible}
\ccc{CopyConstructible}
\ccc{Assignable}

\ccTypes
  \ccNestedType{ECM}{The type of the surface to simplify. Must be a \ccc{EdgeCollapsableMesh}.}{}
\ccGlue    
  \ccNestedType{FT}{A field type that represents the collapse cost.}{}
\ccGlue    
  \ccTypedef{typename halfedge_graph_traits<ECM>::Point Point ;}{The point type of the surface vertex}

\ccCreation
\ccCreationVariable{c}  %% choose variable name

\ccConstructor{CostAndPlacementCache();}{Default constructor}
\ccGlue
\ccConstructor{CostAndPlacementCache( boost::optional<FT> const& cost, boost::optional<Point> const& placement );}
{Constructs a full cache storing the specified \ccc{cost} and \ccc{placement}}

\ccAccessFunctions
  \ccMethod{boost::optional<FT> cost() const;}{The stored cost.}
\ccGlue
  \ccMethod{boost::optional<Point> placement() const;}{The stored placement.}
  
\ccHasModels
\ccRefIdfierPage{CGAL::Surface_mesh_simplification::Cost_and_placement_cache<ECM>}.

\end{ccRefConcept}
