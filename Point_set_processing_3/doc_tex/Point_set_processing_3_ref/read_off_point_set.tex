\ccDefinition

\ccc{CGAL::read_off_point_set()} reads points (positions + normals, if available) from a .off file (ASCII).
Faces are ignored.

\ccInclude{CGAL/read_off_point_set.h}

\ccExample

\begin{ccExampleCode}
typedef CGAL::Exact_predicates_inexact_constructions_kernel Kernel;
typedef CGAL::Point_with_normal_3<Kernel> Point_with_normal;
std::deque<Point_with_normal> points;
char* filename = ...;

// Read the point set file in points[]
if(!CGAL::read_off_point_set(filename, std::back_inserter(points)))
{
  std::cerr << "Error: cannot read file " << filename << std::endl;
  return EXIT_FAILURE;
}
\end{ccExampleCode}
