\begin{ccRefFunction}{read}

\ccDefinition

The function \ccRefName\ reads a given arrangement from a given input stream
using a specific input format.
   
%%%%

\ccInclude{CGAL/IO/Arr_iostream.h}

\ccGlobalFunction{template<typename Traits, typename Dcel, typename Formatter>
            std::istream& read (Arrangement_2<Traits,Dcel>& arr,
                                std::istream& is,
                                Formatter& formatter);}
Reads the arrangement object \ccc{arr} from the given input stream \ccc{is}
using a specific input format defined by \ccc{}formatter.

%%%%

\ccInclude{CGAL/IO/Arr_with_history_iostream.h}

\ccGlobalFunction{template<typename Traits, typename Dcel,
                           typename WithHistoryFormatter>
            std::istream& read (Arrangement_with_history_2<Traits,Dcel>& arr,
                                std::istream& is,
                                WithHistoryFormatter& formatter);}
Reads the arrangement-with-history object \ccc{arr} from the given 
input stream \ccc{is} using a specific input format defined by
\ccc{}formatter.

\ccHeading{Requirements}
\begin{itemize}
\item The instantiated \ccc{Formatter} class must model the
  \ccc{ArrangementInputFormatter} concept.
\item The instantiated \ccc{WithHistoryFormatter} class must model the
  \ccc{ArrangementWithHistoryInputFormatter} concept.
\end{itemize}
