\begin{ccPkgDescription}{Approximation of Ridges and Umbilics on
    Triangulated Surface Meshes \label{Pkg:Ridges_3}}
  \ccPkgHowToCiteCgal{pc-arutsm-07} 

  \ccPkgSummary{Global features related to curvature extrema encode
    important informations used in segmentation, registration,
    matching and surface analysis. Given pointwise estimations of
    local differential quantities on a triangulated surface mesh, this
    package enables the approximation of lines~: ridges or crests, and
    points~: umbilics.}

  \ccPkgDependsOn{Solvers as \ccThirdPartyLapack\ and
    \ccThirdPartyBlas.}
