\ccDefinition

The function \ccRefName\ creates a straight skeleton in the interior of a 2D polygon with holes.

\ccInclude{CGAL/create_straight_skeleton_2.h}

\ccFunction
{template<class PointIterator, class WeightIterator, class HoleIterator, class WeightSequenceIterator, class NT, class K>
boost::shared_ptr< Straight_skeleton_2<K> >
create_straight_skeleton_2 ( PointIterator              outer_contour_vertices_begin
                           , PointIterator              outer_contour_vertices_end
                           , WeightIterator             outer_contour_weights_begin
                           , WeightIterator             outer_contour_weights_end
                           , HoleIterator               holes_begin
                           , HoleIterator               holes_end
                           , WeightSequenceIterator     holes_weights_begin
                           , WeightSequenceIterator     holes_weights_end
                           , boost::optional<NT> const& max_time = boost::optional<NT>()  
                           , K const&                   k        = Exact_predicates_inexact_constructions_kernel          
                          ) ;
}
{returns a new \ccc{Straight_skeleton_2<K>} for the 2D polygon with holes whose outer boundary is given by the point sequence \ccc{[outer_contour_vertices_begin,outer_contour_vertices_end]}
and its holes given by \ccc{[holes_begin,holes_end]}. The weights for the outer boundary is given by the sequence \ccc{[outer_contour_weights_begin,outer_contour_weights_end]},
and the weights for the holes are given by the sequence \ccc{holes_weights_begin,holes_weights_end}. \ccc{max_time} indicates the maximum time for the farthest straight skeleton node.
}

\ccFunction
{template<class PointIterator, class HoleIterator, class NT, class K>
boost::shared_ptr< Straight_skeleton_2<K> >
create_straight_skeleton_2 ( PointIterator              outer_contour_vertices_begin
                           , PointIterator              outer_contour_vertices_end
                           , HoleIterator               holes_begin
                           , HoleIterator               holes_end
                           , NT                         weight
                           , boost::optional<NT> const& max_time = boost::optional<NT>()  
                           , K const&                   k        = Exact_predicates_inexact_constructions_kernel          
                          ) ;
}
{returns a new \ccc{Straight_skeleton_2<K>} for the 2D polygon with holes whose outer boundary is given by the point sequence \ccc{[outer_contour_vertices_begin,outer_contour_vertices_end]}
and its holes given by \ccc{[holes_begin,holes_end]}. The weight for all edges is given by \ccc{weight}. \ccc{max_time} indicates the maximum time for the farthest straight skeleton node.
}

\ccFunction
{template<class PointIterator, class WeightIterator, class NT, class K>
boost::shared_ptr< Straight_skeleton_2<K> >
create_straight_skeleton_2 ( PointIterator              outer_contour_vertices_begin
                           , PointIterator              outer_contour_vertices_end
                           , WeightIterator             outer_contour_weights_begin
                           , WeightIterator             outer_contour_weights_end
                           , boost::optional<NT> const& max_time = boost::optional<NT>()  
                           , K const&                   k        = Exact_predicates_inexact_constructions_kernel          
                          ) ;
}
{returns a new \ccc{Straight_skeleton_2<K>} for the 2D polygon whose outer boundary is given by the point sequence \ccc{[outer_contour_vertices_begin,outer_contour_vertices_end]}.
The weights for the outer boundary is given by the sequence \ccc{[outer_contour_weights_begin,outer_contour_weights_end]}.
\ccc{max_time} indicates the maximum time for the farthest straight skeleton node.
}

\ccFunction
{template<class PointIterator, class NT, class K>
boost::shared_ptr< Straight_skeleton_2<K> >
create_straight_skeleton_2 ( PointIterator              outer_contour_vertices_begin
                           , PointIterator              outer_contour_vertices_end
                           , NT                         weight
                           , boost::optional<NT> const& max_time = boost::optional<NT>()  
                           , K const&                   k        = Exact_predicates_inexact_constructions_kernel          
                          ) ;
}
{returns a new \ccc{Straight_skeleton_2<K>} for the 2D polygon whose outer boundary is given by the point sequence \ccc{[outer_contour_vertices_begin,outer_contour_vertices_end]}.
The weight for all edges is given by \ccc{weight}. \ccc{max_time} indicates the maximum time for the farthest straight skeleton node.
}

\end{enumerate}
