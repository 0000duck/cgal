
\begin{ccPkgDescription}{Combinatorial Maps\label{Pkg:CombinatorialMaps}}
  \ccPkgHowToCiteCgal{cgal:d-cm-11b} \ccPkgSummary{This package
    implements Combinatorial Maps in $d$-dimension. A combinatorial
    map is a mathematical model allowing to represent an object in any
    dimension by describing the subdivision of the object in vertices,
    edges, faces, volumes, ... and the incidence and adjacency
    relationships on them.

    In 2D, a combinatorial map is equivalent to a halfedge data
    structure.}
