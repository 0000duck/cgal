\begin{ccRefClass} {Gmpzf}
\label{Gmpzf}

\ccDefinition

This is an exact floating-point type; it can represent numbers
of the form $m*2^e$, where $m$ is an arbitrary precision integer 
based on the {\sc Gnu} Multiple Precision Arithmetic Library,
and $e$ is of type \ccc{long}. This number type offers funtionality
very similar to \ccc{MP_Float} but is generally faster.

\ccInclude{CGAL/Gmpzf.h}

\ccIsModel
\ccc{EuclideanRing}\\
\\
\ccc{RealEmbeddable}

\ccCreation
\ccCreationVariable{f}

\ccConstructor{Gmpzf();}
             {creates a multiple precision floating-point number
               initialized with $0$.}

\ccHidden \ccConstructor{Gmpzf(const Gmpzf& q);}
            {copy constructor.}

\ccConstructor{Gmpzf(int i);}
            {creates a multiple-precision floating-point number 
              initialized with
             \ccStyle{i}.}

\ccConstructor{Gmpzf(Gmpzf i);}
            {creates a multiple-precision floating-point number 
              initialized with
             \ccStyle{i}.}

\ccConstructor{Gmpzf(double d);}
            {creates a multiple-precision  floating-point number
              initialized with \ccStyle{d}.}
\ccOperations

\ccFunction{std::ostream& operator<<(std::ostream& out, const Gmpzf& f);}
{writes a double approximation of \ccc{f} to the ostream \ccc{out}.}

\ccFunction{std::ostream& print (std::ostream& out, const Gmpzf& a);}
{writes an exact representation of \ccc{f} to the ostream \ccc{out}.}

\ccFunction{std::istream& operator>>(std::istream& in, Gmpzf& f);}
{reads a \ccc{double} from \ccc{in}, then converts it to a \ccc{Gmpzf}.}

\ccImplementation
The significand $m$ of a \ccc{Gmpzf} is reference counted.

\end{ccRefClass}
