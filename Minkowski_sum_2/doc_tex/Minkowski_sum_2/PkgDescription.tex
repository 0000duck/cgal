\begin{ccPkgDescription}{2D Minkowski Sum\label{Pkg:MinkowskiSum2}}
\ccPkgHowToCiteCgal{cgal:w-rms2-06}
\ccPkgSummary{
This package consists of the implementation of the Minkowski sum operation
of two planar polygons and of a planar polygon and a disk (an operation
also known as \emph{offsetting} or \emph{dilating} a polygon).}

\ccPkgDependsOn{\ccRef[2D Arrangements]{Pkg:Arrangement2}}
\ccPkgIntroducedInCGAL{3.3}

\end{ccPkgDescription}
