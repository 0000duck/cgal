\ccMethod{result_type operator()(const Arc_2& cv1, const Arc_2& cv2, CGAL::Arr_curve_end ce) const;}
{
Compare the y-coordinates of 2 arcs at their ends near the left or right boundary of the parameter space.
}
\ccGlue
\begin{description}
\item[Parameters:]
\begin{description}
\item[cv1]the first arc. \item[cv2]the second arc. \item[ce]the arc end indicator. \end{description}
\end{description}
\begin{description}
\item[Returns:]\ccc{CGAL::SMALLER} is y(cv1,ce) $<$ y(cv2,ce2) near left/right boundary \ccc{CGAL::EQUAL} is y(cv1,ce) = y(cv2,ce2) near left/right boundary, i.e., they overlap
\ccc{CGAL::LARGER} is y(cv1,ce) $>$ y(cv2,ce2) near left/right boundary\end{description}
\begin{description}
\item[Precondition:]the ce ends of the arcs cv1 and cv2 lie either on the left boundary or on the right boundary of the parameter space. \end{description}
\ccGlue
