\begin{ccRefClass} {Gmpzf}
\label{Gmpzf}

\ccDefinition

This is an multiple-precision floating-point type; it can represent
numbers of the form $m*2^e$, where $m$ is an arbitrary precision integer
based on the {\sc Gnu} Multiple Precision Arithmetic Library, and $e$
is of type \ccc{long}. This type can be considered exact, even if the
exponent is not a multiple-precision number.  This number type offers
functionality very similar to \ccc{MP_Float} but is generally faster.

\ccInclude{CGAL/Gmpzf.h}

\ccIsModel
\ccc{EuclideanRing}\\
\\
\ccc{RealEmbeddable}

\ccCreation
\ccCreationVariable{f}

\ccConstructor{Gmpzf();}
             {creates a \ccc{Gmpzf} initialized with $0$.}

\ccHidden \ccConstructor{Gmpzf(const Gmpzf& q);}
            {copy constructor.}

\ccConstructor{Gmpzf(int i);}
            {creates a \ccc{Gmpzf} initialized with \ccStyle{i}.}

\ccConstructor{Gmpzf(long int l);}
            {creates a \ccc{Gmpzf} initialized with \ccStyle{l}.}

\ccConstructor{Gmpzf(const Gmpz& i);}
            {creates a \ccc{Gmpzf} initialized with \ccStyle{i}.}

\ccConstructor{Gmpzf(const Gmpfr& f);}
            {creates a \ccc{Gmpzf} initialized with \ccStyle{f}.}

\ccConstructor{Gmpzf(double d);}
            {creates a \ccc{Gmpzf} initialized with \ccStyle{d}.}
\ccOperations

\ccFunction{std::ostream& operator<<(std::ostream& out, const Gmpzf& f);}
{writes a double approximation of \ccc{f} to the ostream \ccc{out}.}

\ccFunction{std::ostream& print (std::ostream& out, const Gmpzf& f);}
{writes an exact representation of \ccc{f} to the ostream \ccc{out}.}

\ccFunction{std::istream& operator>>(std::istream& in, Gmpzf& f);}
{reads a \ccc{double} from \ccc{in}, then converts it to a \ccc{Gmpzf}.}

\ccImplementation
The significand $m$ of a \ccc{Gmpzf} is a \ccc{Gmpz} and is reference
counted. The exponent $e$ of a \ccc{Gmpzf} is a \verb=long=.

\end{ccRefClass}
