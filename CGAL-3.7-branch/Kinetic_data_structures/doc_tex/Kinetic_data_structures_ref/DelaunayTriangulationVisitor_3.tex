\begin{ccRefConcept}{Kinetic::DelaunayTriangulationVisitor_3}

\ccDefinition

This concept is for proxy objects which get notified when a kinetic Delaunay triangulation changes.

\ccCreationVariable{v}

\ccOperations

\ccMethod{void pre_remove_vertex(Vertex_handle);}{The vertex is about to be deleted.}

\ccMethod{void post_remove_vertex(Point_key, Cell_handle);}{The vertex was just removed.}

\ccMethod{void pre_insert_vertex(Point_key, Cell_handle);}{The vertex is about to be inserted into the cell (although the \ccc{Cell_handle} might be \ccc{NULL}).}

\ccMethod{void post_insert_vertex(Vertex_handle);}{The vertex was just inserted.}

\ccMethod{void change_vertex(Vertex_handle);}{The trajectory of the point at the vertex changed.}

\ccMethod{void create_cell(Cell_handle);}{The cell was just created and initialized.}

\ccMethod{void destroy_cell(Cell_handle);}{The cell is about to be removed.}

\ccMethod{void pre_edge_flip(Edge);}{The edge is about to be flipped.}

\ccMethod{void post_edge_flip(Facet);}{The facet was just created with a flip.}

\ccMethod{void pre_facet_flip(Facet);}{The facet is about to be flipped.}

\ccMethod{void post_facet_flip(Edge);}{The edge was just created with a flip.}


\ccHasModels 

\ccc{Kinetic::Delaunay_triangulation_visitor_base_3}, \ccc{Kinetic::Delaunay_triangulation_recent_edges_visitor_3<Triangulation>}, \ccc{Kinetic::Delaunay_triangulation_event_log_visitor_3}

\end{ccRefConcept}
