\begin{ccRefClass}[Nef_polyhedron_3<Traits>::]{Halffacet_cycle_iterator}

\ccDefinition

The type \ccc{Halffacet_cycle_iterator} iterates over a list of 
\ccc{Object_handles}. Each item of that list can either be assigned
to \ccc{SHalfedge_handle} or \ccc{SHalfloop_handle}. To find out which
of these assignment works out, the member functions \ccc{is_shalfedge()}
and \ccc{is_shalfloop()} are provided.

\ccInclude{CGAL/Nef_polyhedron_3.h}

\ccTypes
\ccThree{Halfedge_const_handle;}{SHalfloop_handle(hfc);;}{}
\ccThreeToTwo

\ccNestedType{SHalfedge_handle}{const handle to SHalfedge.}
\ccGlue
\ccNestedType{SHalfloop_handle}{const handle to SHalfloop.}

\ccCreation
\ccCreationVariable{hfc}

\ccConstructor{Halffacet_cycle_iterator();}{default constructor.}

\ccOperations

\ccMethod{bool is_shalfedge() const;}
{returns true if \ccVar\ represents a \ccc{SHalfedge_handle}.}
\ccGlue
\ccMethod{bool is_shalfloop() const;}
{returns true if \ccVar\ represents a \ccc{SHalfloop_handle}.}

\ccMethod{operator SHalfedge_handle() const;}
{casts \ccVar\ to \ccc{SHalfedge_handle}.}
\ccGlue
\ccMethod{operator SHalfloop_handle() const;}
{casts \ccVar\ to \ccc{SHalfloop_handle}.}

\end{ccRefClass}
