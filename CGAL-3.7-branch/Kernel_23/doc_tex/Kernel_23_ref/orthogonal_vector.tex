\begin{ccRefFunction}{orthogonal_vector}

\ccFunction{Vector_3<Kernel> orthogonal_vector( const Plane_3<Kernel>& p);}
 {computes an orthogonal vector of the plane \ccc{p}, which is directed to 
  the positive side of this plane.}

\ccFunction{Vector_3<Kernel> orthogonal_vector( const Point_3<Kernel>& p,
                                           const Point_3<Kernel>& q,
                                           const Point_3<Kernel>& r );}
 {computes an orthogonal vector of the plane defined by \ccc{p}, \ccc{q} and \ccc{r},
  which is directed to the positive side of this plane.}


\end{ccRefFunction}
