\begin{ccRefFunction}{lexicographically_smaller_or_equal}
\ccInclude{CGAL/predicates_d.h}

\ccFunction{bool lexicographically_smaller_or_equal( const Point_d<R>&
  p, const Point_d<R>& q);}{returns \ccc{true} iff $p$ is
  lexicographically smaller than $q$ with respect to Cartesian
  lexicographic order of points or equal to $q$.  \ccPrecond
  \ccc{p.dimension() == q.dimension()}.}

\end{ccRefFunction}
