\begin{ccRefConcept}{ApolloniusGraphHierarchyVertexBase_2}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries
\ccCreationVariable{hvb}


\ccDefinition 
The vertex of an Apollonius graph
included in an Apollonius graph hierarchy has to provide
some pointers to the corresponding vertices in the
graphs of the next and preceeding levels.
Therefore, the concept \ccc{ApolloniusGraphHierarchyVertexBase_2}
extends the concept \ccc{ApolloniusGraphVertexBase_2}, by
adding two \ccc{void*} pointers to the correponding vertices in the
next and previous level graphs.


\ccRefines
\ccc{ApolloniusGraphVertexBase_2}


\ccTypes
\ccc{ApolloniusGraphHierarchyVertexBase_2} does not introduce any
types in addition to those of \ccc{ApolloniusGraphVertexBase_2}.

constructor.}
\ccGlue

  embedded at the center of \ccc{p}.}
\ccGlue
\ccConstructor

\ccOperations

\ccGlue 
\ccMethod{void set_down(void *d);}{Sets the pointer to the corresponding vertex
