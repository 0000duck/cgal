\begin{ccRefConcept}{VoronoiTraits_2}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries
\ccDefinition

The concept \ccc{VoronoiTraits_2} defines the combinatorial and
geometric predicate requirements for the
\ccc{Voronoi_diagram_2<DG,VT,AP>} class. The
\ccc{VoronoiTraits_2} concept essentially defines the requirements on
the predicates that determine whether a feature of the triangulated
Delaunay graph has a degenerate dual feature in the Voronoi diagram
that is the dual of the triangulated Delaunay graph.

\ccRefines
\ccc{DefaultConstructible}, \ccc{CopyConstructible}, \ccc{Assignable}

\ccGlue
\ccNestedType{Site_2}{A type for the sites of the Voronoi diagram.}
\ccGlue
\ccNestedType{Delaunay_graph}{A type for the triangulated Delaunay
  graph. The type \ccc{Delaunay_graph} must be a model of the
  \ccc{DelaunayGraph_2} concept.}

\ccTypedef{typedef Delaunay_graph::Face_handle Face_handle;}{The type
  of the face handles of the Delaunay graph}
\ccGlue
\ccTypedef{typedef Delaunay_graph::Vertex_handle Vertex_handle;}{The
  type of the vertex handles of the Delaunay graph.}

\ccTwo{VoronoiDiagram_2::Construct_Voronoi_point_2}{}
\ccNestedType{Access_site_2}{A type for a functor that accesses the
  site associated with a vertex. The functor should be a model of the
  concepts \ccc{DefaultConstructible}, \ccc{CopyConstructible},
  \ccc{Assignable} and \ccc{AdaptableFunctor} (with one argument). The
  functor must provide the following operator:\\
  \centerline{\ccc{result_type operator()(Vertex_handle v)}}\\
  where the result type \ccc{result_type} must be either \ccc{Site} or
  \ccc{const Site_2&}.}
\ccGlue
\ccNestedType{Construct_Voronoi_point_2}{A type for a functor that
  constructs the dual point of a (triangular) face in the Delaunay
  graph. This point is the Voronoi vertex of the three sites defining
  the face in the Delaunay graph. The functor must be a model of the
  concepts \ccc{DefaultConstructible}, \ccc{CopyConstructible},
  \ccc{Assignable}, \ccc{AdaptableFunctor} (with one argument). It
  must provide the following operator:\\
  \centerline{\ccc{Point_2 operator()(Face_handle f)}}.
  The face handle \ccc{f} must not correspond to an infinite face.}
\ccGlue
\ccNestedType{Has_nearest_site_2}{A tag for determining if the Voronoi
  traits class provides a functor for performing nearest site queries.
  This tag is equal to either \ccc{CGAL::Tag_true} (a nearest site
  query functor is available) or \ccc{CGAL::Tag_false} (a nearest site
  query functor is not available).}
\ccGlue
\ccNestedType{Nearest_site_2}{A type for a functor that performs
  nearest site queries. Semantically, the result of the query is
  either a face, edge or vertex of the Delaunay graph. It is a face if
  the query point has at least three closest sites; the returned face
  has closest sites as vertices. It is an edge if the query point is
  equidistant to exactly two vertices of the Delaunay graph, which are
  the source and target vertices of the edge. In all other cases, the
  search result is a vertex, namely, the unique vertex of the 
  Delaunay graph closest to the query point. The functor must be a
  model of the concepts \ccc{DefaultConstructible},
  \ccc{CopyConstructible}, \ccc{Assignable}, \ccc{AdaptableFunctor}
  (with two arguments). It must provide the following operator:\\
  \centerline{\ccc{result_type operator()(Delaunay_graph dg, Point_2 p)}}\\
  where the result type \ccc{result_type} is
  \ccc{boost::variant<Vertex_handle,Edge,Face_handle>}.\\
  This type is required only if
  \ccc{Has_nearest_site_2} is equal to \ccc{CGAL::Tag_true}.}
