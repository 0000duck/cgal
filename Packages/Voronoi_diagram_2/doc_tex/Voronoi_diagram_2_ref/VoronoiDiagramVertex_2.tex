\begin{ccRefConcept}{VoronoiDiagramVertex_2}

\ccDefinition
%
The concept \ccc{VoronoiDiagramVertex_2} defines the requirements of the
vertex provided by the class \ccc{Voronoi_diagram_2}.

\ccRefines
\ccc{DefaultConstructible}\\
\ccc{CopyConstructible}\\
\ccc{Assignable}\\
\ccc{EqualityComparable}\\
\ccc{LessThanComparable}

\ccTypes
%
\ccTwo{VoronoiDiagramVertex_2::Halfedge_around_vertex+}{}
\ccNestedType{Halfedge}{A type for the halfedges of the Voronoi diagram.}
\ccGlue
\ccNestedType{Face}{A type for the faces of the Voronoi diagram.}
\ccGlue
\ccNestedType{Vertex_handle}{Handle for the vertices of the Voronoi diagram.}
\ccGlue
\ccNestedType{Face_handle}{Handle for the faces of the Voronoi diagram.}
\ccGlue
\ccNestedType{Halfedge_handle}{Handle for the halfedges of the Voronoi
  diagram.}
\ccGlue
\ccNestedType{Point_2}{A type for the point represented by the
  vertex.}
\ccGlue
\ccNestedType{size_type}{A type for sizes.}
\ccGlue
\ccNestedType{Halfedge_around_vertex_circulator}{A type for a
  circulator that allows to traverse all incident halfedges. The value
  type of the circulator is \ccc{Halfedge}.}
%
\ccThree{typedef typename Delaunay_graph::Face_handle}{Delaunay_face_handle;}{}
\ccThreeToTwo
\ccNestedType{Delaunay_graph}{A type for the Delaunay graph. It must be a
  model of the \ccc{DelaunayGraph_2} concept.}
\ccGlue
\ccTypedef{typedef typename Delaunay_graph::Face_handle Delaunay_face_handle;}
{A type for the handle of the dual face.}
\ccGlue
\ccTypedef{typedef typename Delaunay_graph::Vertex_handle
Delaunay_vertex_handle;}
{A type for the handles of vertices in the Delaunay graph.}

\ccCreationVariable{v}  %% choose variable name

\ccHeading{Access Methods}
%\ccThree{Dual_face_handle}{v.dual_face()+}{}
\ccThree{Halfedge_around_vertex_circulator}{v.incident_halfedges()+}{}
\ccMethod{Halfedge_handle halfedge();}{Returns an incident halfedge
  that has \ccc{v} as its target.}
\ccGlue
\ccMethod{Halfedge_around_vertex_circulator incident_halfedges();}
{Returns a circulator that allows the traversal of the halfedges that
  have \ccc{v} as their target. The edges are traversed around \ccc{v}
  in a order opposite to that of the \ccc{Ccb_halfedges_circulator}.}
\ccGlue
\ccMethod{size_type degree();}{Returns the degree of the vertex,
  i.e. the number of incident halfedges.}
\ccGlue
\ccMethod{Point_2 point();}{Returns the point represented by the vertex.}
\ccGlue
\ccMethod{Delaunay_face_handle dual();}
{Returns a handle to the dual face, i.e., the corresponding dual face in the
  Delaunay graph.}
\ccGlue
\ccMethod{Delaunay_vertex_handle first();}{returns a handle to the vertex in
the Delaunay graph corresponding to the first generating site of the
Voronoi vertex.}
\ccGlue
\ccMethod{Delaunay_vertex_handle second();}{returns a handle to the vertex
in the Delaunay graph corresponding to the second generating site of the
Voronoi vertex.}
\ccGlue
\ccMethod{Delaunay_vertex_handle third();}{returns a handle to the vertex
in the Delaunay graph corresponding to the third generating site of the
Voronoi vertex.}
\ccGlue
\ccMethod{Delaunay_vertex_handle operator[](unsigned int i);}
{returns a handle to the vertex in the Delaunay graph corresponding to 
the $(i+1)$-th generating site of the Voronoi vertex.
\ccPrecond{\ccc{i} must be smaller or equal to 2.}}

\ccHeading{Predicate Methods}

\ccThree{bool}{e.is_incident_edge(Halfedge_handle e)+}{}
%
\ccMethod{bool is_incident_edge(Halfedge_handle e);}{Returns \ccStyle{true}
  if the halfedge \ccc{*e} is incident to \ccc{v}.}
\ccGlue
\ccMethod{bool is_incident_face(Face_handle e);}{Returns \ccStyle{true}
  if the face \ccc{*f} is incident to \ccc{v}.}
\ccGlue
\ccMethod{bool is_valid();}{Returns \ccStyle{true} if the following
  conditions are met: the dual face is not an infinite face; all
  incident halfedges have the vertex as their target.}

\ccHasModels
\ccc{CGAL::Voronoi_diagram_2<DG,VT>::Vertex}\\

\ccSeeAlso
\ccc{VoronoiDiagramHalfedge_2}\\
\ccc{VoronoiDiagramFace_2}\\
\ccc{DelaunayGraph_2}\\
\ccc{CGAL::Voronoi_diagram_2<DG,VT>}
\end{ccRefConcept}
