\begin{ccRefConcept}{VoronoiDiagramVertex_2}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries
\ccDefinition

The concept \ccc{VoronoiDiagramVertex_2} defines the requirements of the
vertex provided by the Voronoi diagram adaptor. It refines
the \ccc{PlanapMapDcelVertex_2} concept (the non-mutable part), by
adding some additional requirements specific to Voronoi diagrams.

\ccRefines
\ccc{DefaultConstructible}\\
\ccc{CopyConstructible}\\
\ccc{Assignable}\\
\ccc{PlanarMapDcelVertex_2}

\ccTypes
The \ccc{VoronoiDiagramVertex_2} concept introduces the following
types in addition to those of the \ccc{PlanarMapDcelVertex_2} concept.

\ccThree{typedef typename Dual_graph::Face_handle}{Dual_face_handle;}{}
\ccThreeToTwo
\ccNestedType{Dual_graph}{A type for the Delaunay graph. It must be a
  model of the \ccc{DelaunayGraph_2} concept.}
\ccTypedef{typedef typename Dual_graph::Face_handle Dual_face_handle;}
{A type for the handle of the dual face.}

\ccCreationVariable{v}  %% choose variable name

\ccThree{Dual_face_handle}{v.has_source()+}{}
\ccThreeToTwo

\ccHeading{Access Methods}
\ccMethod{Dual_face_handle dual_face();}{returns the dual face, i.e., the
  corresponding dual face in the Delaunay graph.}

\ccHeading{Predicate Methods}

\ccMethod{bool is_valid();}{returns \ccStyle{true} if the following
  conditions are met: the dual face is not an infinite face; all
  incident halfedges have the vertex as their target.}

\ccHasModels
\ccc{CGAL::Voronoi_diagram_adaptor_2<DG,VT>::Vertex}\\

\ccSeeAlso
\ccc{VoronoiDiagramHalfedge_2}\\
\ccc{VoronoiDiagramFace_2}\\
\ccc{CGAL::Voronoi_diagram_adaptor_2<DG,VT>}
\end{ccRefConcept}
