\begin{ccRefConcept}{DelaunayGraphLocateType_2}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries
\ccDefinition

The concept \ccc{DelaunayGraphLocateType_2} defines the requirements of the
\ccc{Locate_type} type defined in the \ccc{DelaunayGraphPointLocator_2}
concept. Essentially it provides the requirements for a class
representing the return type of a point location query performed on a
Delaunay graph. Semantically, the locate type is either a face, edge or
vertex of the Delaunay graph. It is a face if the query point is
equidistant to at least three vertices of the Delaunay graph, namely
the three vertices of the returned face. It is an edge if the query
point is equidistant to exactly two vertices of the Delaunay graph,
which are the source and target vertices of the edge. In all other
cases, the locate type is a vertex, namely, the unique vertex of the
Delaunay graph closest to the query point. The
\ccc{DelaunayGraphLocateType_2} concept encapsulates the semantics
described above.

\ccHeading{Predicate Methods}
\ccThree{bool}{lt.is_vertex()+}{}
\ccThreeToTwo
%
\ccMethod{bool is_vertex();}{Returns \ccStyle{true} iff the result of
  the point location corresponds to a vertex of the Delaunay graph,
  i.e., iff the query point is closest to a single vertex of the
  Delaunay graph.}
\ccGlue
\ccMethod{bool is_edge();}{Returns \ccStyle{true} iff the result of
  the point location corresponds to an edge of the Delaunay graph,
  i.e., iff the query point is closest to exactly two vertices of the
  Delaunay graph, which are the source and target vertex of the
  afore-mentioned edge of the Delaunay graph.}
\ccGlue
\ccMethod{bool is_face();}{Returns \ccStyle{true} iff the result of
  the point location corresponds to a face of the Delaunay graph,
  i.e., iff the query point is closest to at least three vertices of
  the Delaunay graph, which are three vertices of the afore-mentioned
  face of the face of the Delaunay graph.}

\ccMethod{Vertex_handle vertex();}{Returns a handle to the vertex of
  the Delaunay graph which is the result of the point location query.

\ccGlue
\ccMethod{Edge edge();}{Returns the edge of the Delaunay graph which is the
  result of the point location query.

\ccMethod{Face_handle face();}{Returns a handle to the face of the
  Delaunay graph which is the result of the point location query.
\ccPrecond{\ccc{is_face()} must be \ccStyle{true}.}}
