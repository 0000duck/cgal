\begin{ccRefFunctionObjectConcept}{FaceDegeneracyTester}

\ccRefines
\ccc{DefaultConstructible}, \ccc{CopyConstructible}, \ccc{Assignable},
\ccc{AdaptableFunctor} (with two arguments)

\ccTypes

In addition to the types required by the \ccc{AdaptableFunctor}
concept, a model of this concept must provide the following types:

%\ccTwo{FaceDegeneracyTester::Dual_graph+}{}
%
\ccThree{typedef Delaunay_graph::Vertex_handle}{Vertex_handle+}{}
\ccThreeToTwo
\ccNestedType{Delaunay_graph}{A type for the Delaunay graph. It must be
  model of the \ccc{DelaunayGraph_2} concept.}
%
\ccGlue
\ccTypedef{typedef Delaunay_graph::Vertex_handle Vertex_handle;}{}


\ccCreationVariable{fdt}

\ccOperations
%It must also provide the following operation:

\ccThree{bool}{Delaunay_graph dg, fdt(Vertex_handle v)+}{}
\ccThreeToTwo
\ccMemberFunction{bool operator()(Delaunay_graph dg, Vertex_handle v);}
{Returns \ccStyle{true}, if the dual face of \ccc{v} (that
  belongs to the Delaunay graph \ccc{dg}) is
  degenerate, i.e., it has zero area.}
%


\ccHeading{Miscellaneous}
The following methods are important when the face degeneracy tester
maintains a state. This can happen if the face degeneracy tester
caches its results.

\ccThree{bool}{fdt.swap(other)++}{}
%
\ccMethod{void clear();}{Clears the state of the tester.}
\ccGlue
\ccMethod{void swap(FaceDegeneracyTester other);}{The tester
  \ccc{fdt} and \ccc{other} are swapped. This method should be
  preferred to \ccStyle{fdt=other} or \ccStyle{fdt(other)} if
  \ccc{other} is deleted afterwards.}
\ccGlue
\ccMethod{bool is_valid();}{Tests the validity of the tester.}
