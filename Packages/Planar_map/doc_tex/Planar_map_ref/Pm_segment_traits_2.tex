\begin{ccRefClass}{Pm_segment_traits_2<Kernel>}
    
\ccDefinition 
  The interface class \ccStyle{Pm_segment_traits_2<Kernel>} is a model of the
  \ccc{PlanarMapTraits_2} concept. It is provided as part of the package and
  can be used as reference for resolving ambiguities.
  It must be templated with a CGAL Kernel model that is templated in turn
  by an exact number type, in order to avoid robustness problems, 
  although other number types could be used at the user's own risk.
  For example, \ccStyle{Cartesian<leda_rational>},
  \ccStyle{Homogeneous<leda_integer>},
  \ccStyle{leda_rat_kernel_traits}.

\ccInclude{CGAL/Pm_segment_traits_2.h}
 
\ccIsModel
    \ccc{PlanarMapTraits_2}

\end{ccRefClass} % Pm_segment_traits_2
