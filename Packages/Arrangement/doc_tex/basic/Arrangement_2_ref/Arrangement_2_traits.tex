\begin{ccRefConcept}{ArrangementTraits_2}

\ccDefinition
A model for the\ccRefName\ concept aggregates the geometric types and
primitive operations used in the arrangement
\ccc{Arrangement_2<Dcel,Traits,Base_node>} data structure.

Note that the concept \ccRefName\ refines the concept
\ccc{PlanarMapWithIntersectionsTraits_2} and inherits all its types and
operations.

In addition to the requirements of the \ccc{PlanarMapWithIntersectionsTraits_2}
concept, it must provide the types and operations listed below.

\ccRefines
\ccc{PlanarMapWithIntersectionsTraits_2}
\lcTex{(\ccRefPage{PlanarMapWithIntersectionsTraits_2})}.

\ccCreationVariable{arr_traits}

\ccCreation
    
\ccConstructor{Traits();}
{A default constructor.}

\ccOperations
\ccMethod{X_monotone_curve_2 curve_opposite
	                           (const X_monotone_curve_2& cv) const;}
         {flips the direction of the curve \ccc{cv}. The source and target
          points of \ccc{cv} interchange.}

\ccHasModels

  \ccc{Arr_segment_cached_traits_2<Kernel>}\\
  \ccc{Arr_polyline_traits_2<Segment_traits>} \\
  \ccc{Arr_conic_traits_2<Kernel>}

\end{ccRefConcept}
