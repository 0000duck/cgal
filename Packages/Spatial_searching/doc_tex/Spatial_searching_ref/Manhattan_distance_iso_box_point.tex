\ccDefinition
  
The class \ccRefName\ provides an implementation of
the \ccc{GeneralDistance} concept
for the Manhattan distance ($l_1$ metric)
between a $d$-dimensional iso-box and a $d$-dimensional point 
and the Manhattan distance between a $d$-dimensional iso-box
and a $d$-dimensional iso-box defined as a $k$-$d$ tree rectangle.

\ccInclude{CGAL/Manhattan_distance_iso_box_point.h}

\ccParameters
Expects for the first template argument
a model for the concept \ccc{Traits}, for example \ccc{CGAL::Kd_tree_traits_3<Kernel>}.

Expects for the second template argument a model
of the concept \ccc{IsoBox},
for example \ccc{CGAL::Cartesian<double>::Iso_cuboid_3}.


\ccIsModel

\ccc{GeneralDistance}

\ccTypes

\ccTypedef{Traits::FT FT;}{Number type.}
\ccTypedef{Traits::Point_d Point_d;}{Point type.}
\ccTypedef{IsoBox Query_item;}{Query item type.}

\ccCreation
\ccCreationVariable{d}  %% choose variable name

\ccConstructor{Manhattan_distance_iso_box_point();}{Default constructor.}

\ccOperations

\ccMethod{FT transformed distance(IsoBox b, Point_d p);}{Returns the transformed 
distance between \ccc{b} and \ccc{p}.}

\ccMethod{FT transformed_distance(FT d);} {Returns $d$.}

\ccMethod{FT inverse_of_transformed_distance(FT d);} {Returns $d$.}

\ccMethod{FT min_distance_to_rectangle(IsoBox b, Kd_tree_rectangle<Traits> r);}
{Returns the minimal distance between a point from \ccc{b} and a point from
\ccc{r}.}

\ccMethod{FT max_distance_to_rectangle(IsoBox b, Kd_tree_rectangle<Traits> r);}
{Returns the maximal distance between the iso-box \ccc{b} and
a point from \ccc{r} furthest to \ccc{b}.}


\ccSeeAlso

\ccc{GeneralDistance}.




\end{ccRefClass}
