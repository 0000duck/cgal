\ccDefinition
  
The class \ccRefName\ provides an implementation of the
\ccc{GeneralDistance} concept for the Euclidean distance ($l_2$
metric) between a $d$-dimensional sphere and a point, and the
Euclidean distance between a $d$-dimensional sphere and a
$d$-dimensional iso-rectangle defined as a $k$-$d$ tree rectangle.


\ccInclude{CGAL/Euclidean_distance_sphere_point.h}

\ccParameters
Expects for the first template argument
an implementation for
$d$-dimensional points of
the concept \ccc{PointTraits}, 
for example \ccc{CGAL::Kd_tree_traits_point_d< Cartesian_d<double> >}.

Expects for the second template argument an implementation 
for $d$-dimensional spheres of the concept \ccc{Sphere},
for example \ccc{CGAL::Cartesian_d<double>::Sphere_d}.


\ccIsModel

GeneralDistance

\ccTypes

\ccTypedef{PointTraits::NT NT;}{Number type.}
\ccTypedef{PointTraits::Point Point;}{Point type.}
\ccTypedef{Sphere Query_item;}{Query item type.}


\ccCreation
\ccCreationVariable{d}  %% choose variable name


\ccConstructor{Euclidean_distance_sphere_point();}
{Default constructor.}

\ccOperations

\ccMethod{NT distance(Sphere s, Point p);}{Returns the distance between \ccc{s} and \ccc{p}.}

\ccMethod{NT min_distance_to_queryitem(Sphere s, Kd_tree_rectangle<PointTraits> r);}
{Returns the minimal distance between a point from the sphere \ccc{s} and a point from
\ccc{r}.}

\ccMethod{NT max_distance_to_queryitem(Sphere s, Kd_tree_rectangle<PointTraits> r);}
{Returns the maximal distance between the sphere \ccc{s} and
a point from \ccc{r} furthest to \ccc{s}. }

\ccMethod{NT transformed_distance(NT d);} {Returns $d^2$.}

\ccMethod{NT inverse_of_transformed_distance(NT d);} {Returns $d^{1/2}$.}


\ccSeeAlso

GeneralDistance.
