\begin{ccRefClass}{L2_distance_sphere_point<QueryItem,Item>}  %% add template arg's if necessary

\ccDefinition

The class \ccRefName\ provides an implementation of \\
the GeneralDistance concept
for the L2 distance between a $d$-dimensional sphere and a point and the L2 distance between a $d$-dimensional sphere
and a $d$-dimensional iso-rectangle defined as a $k$-$d$ tree rectangle.

\ccInclude{CGAL/L2_distance_sphere_point.h}

\ccParameters

Expects for the parameter \ccc{QueryItem}
\ccc{item} an implementation of $d$-dimensional spheres
for example \ccc{CGAL::sphere_d<R>}.
Expects for the parameter \ccc{Item} an implementation
of a $d$-dimensional point, for example \ccc{CGAL::Point_d<R>}.

\ccIsModel

GeneralDistance

\ccTypes

\ccTypedef{Item::R::FT NT;}{Number type.}

\ccCreation
\ccCreationVariable{s}  %% choose variable name

\ccConstructor{L2_distance_sphere_point();}{Default constructor implementing $l_2$ metric for
two-dimensional spheres.}

\ccConstructor{L2_distance_sphere_point(int d);}
{Constructor implementing $l_2$ metric for
$d$-dimensional spheres.}

\ccHeading{Member Functions}

\ccMethod{NT transformed_distance(NT d);} {Returns $d^2$.}

\ccMethod{NT inverse_of_transformed_distance(NT d);} {Returns $\sqrt d$.}

\ccMethod{NT distance(QueryItem& q, Item&p);}{Returns the distance between \ccc{q} and \ccc{p}.}

\ccMethod{NT min_distance_to_queryitem(QueryItem& q, Kd_tree_rectangle<NT>& r;);}
{Returns the minimal distance between a point from the sphere \ccc{q} and a point from
\ccc{r}}

\ccMethod{NT max_distance_to_queryitem(QueryItem& q, Kd_tree_rectangle<NT>& r;);}
{Returns the maximal distance between the sphere \ccc{q} and
a point from \ccc{r} furthest to \ccc{q}.}


\ccSeeAlso

GeneralDistance.
