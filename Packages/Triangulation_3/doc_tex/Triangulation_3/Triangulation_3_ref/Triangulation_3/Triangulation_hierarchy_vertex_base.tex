\begin{ccRefConcept}{Triangulation_hierarchy_vertex_base_3_concept}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries

\ccDefinition 
The vertex of a triangulation included in a triangulation hierarchy has to
provide some pointers to the corresponding vertices in the triangulations of
the next and preceeding levels.  Therefore, the concept
\ccc{Triangulation_hierarchy_vertex_base_3_concept} extends the concept
\ccc{Triangulation_vertex_base_3_concept}, adding two \ccc{void*} pointers to
the correponding vertices in the next and previous level triangulations.


\ccGeneralizes
\ccc{Triangulation_vertex_base_3_concept}


\ccTypes
\ccc{Triangulation_hierarchy_vertex_base_3_concept} does not introduce other
types than those of \ccc{Triangulation_vertex_base_3_concept}.


\ccCreation
\ccCreationVariable{v}  %% choose variable name

\ccConstructor{Triangulation_hierarchy_vertex_base_3_concept();}{default
constructor.}
\ccGlue
\ccConstructor{Triangulation_hierarchy_vertex_base_3_concept(const
Point & p).}{constructs a vertex embedding the point \ccc{p}.}
\ccGlue
\ccConstructor{Triangulation_hierarchy_vertex_base_3_concept(const
Point & p, void* c).}{constructs a vertex embedding the point \ccc{p}
and pointing to cell \ccc{c}.}

\ccOperations
\ccMethod{void* up();}{returns a pointer to the corresponding vertex
(if any) of the next level triangulation;}
\ccGlue
\ccMethod{void* down();}{returns a pointer to the corresponding vertex
 of the previous level triangulation;}
\ccGlue
\ccMethod{void set_up(void *u);}{sets the pointer to the corresponding vertex
of the next level triangulation;} 
\ccGlue 
\ccMethod{void set_down(void *d);}{sets the pointer to the corresponding vertex
of the previous level triangulation;}
