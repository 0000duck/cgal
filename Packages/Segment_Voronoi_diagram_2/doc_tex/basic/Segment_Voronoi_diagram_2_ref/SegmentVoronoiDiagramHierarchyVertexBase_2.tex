\begin{ccRefConcept}{SegmentVoronoiDiagramHierarchyVertexBase_2}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links

\ccCreationVariable{hvb}


\ccDefinition 
The vertex of a segment Voronoi diagram
included in a segment Voronoi diagram hierarchy has to provide
some pointers to the corresponding vertices in the
graphs of the next and preceeding levels.
Therefore, the concept \ccc{SegmentVoronoiDiagramHierarchyVertexBase_2}
refines the concept \ccc{SegmentVoronoiDiagramVertexBase_2}, by
adding two vertex handles to the correponding vertices for the
next and previous level graphs.


\ccRefines
\ccc{SegmentVoronoiDiagramVertexBase_2}


\ccTypes
\ccc{SegmentVoronoiDiagramHierarchyVertexBase_2} does not introduce
any types in addition to those of
\ccc{SegmentVoronoiDiagramVertexBase_2}.

  vertex of the previous level segment Voronoi diagram. If such a
  vertex does not exist \ccc{Vertex_handle()} is returned.}

  vertex of the next level segment Voronoi diagram.}

\ccGlue 
\ccMethod{void set_down(Vertex_handle d);}{Sets the handle for the
  vertex of the previous level segment Voronoi diagram;}
