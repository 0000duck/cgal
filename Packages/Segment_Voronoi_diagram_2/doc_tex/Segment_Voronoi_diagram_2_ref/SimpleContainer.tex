\begin{ccRefConcept}{SimpleContainer}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries
\ccCreationVariable{vb}
\ccDefinition

The concept \ccc{SimpleContainer} describes the requirements for the
point container required by the
\ccc{CGAL::Segment_Voronoi_diagram_2<Gt,PC,DS>},
\ccc{CGAL::Segment_Voronoi_diagram_hierarchy_2<Gt,STag,PC,DS>},
\ccc{CGAL::Segment_Voronoi_diagram_storage_site_2<Gt,PC>} and
\ccc{CGAL::Segment_Voronoi_diagram_vertex_base_2<Gt,PC,SSTag>}
classes. Note that the iterators returned by the \ccc{insert} method
must not be invalidated when insertions and deletions are performed on
the \ccc{SimpleContainer}.

\ccTypes
\ccTwo{SimpleContainer::value_type+}{}
\ccNestedType{value_type}{The value type of the iterator of the container.}
%
\ccGlue
\ccNestedType{iterator}{A type for the iterator of the container.}
%
\ccGlue
\ccNestedType{size_type}{A type for the size of the container.}
%

\ccCreation
\ccCreationVariable{sc}  %% choose variable name
%\ccTwo{SimpleContainer sc;+}{}
%
\ccConstructor{SimpleContainer();}{Default constructor.}



\ccHeading{Methods}
\ccThree{size_type}{insert(value_type item)+}{}
%
\ccMethod{iterator insert(value_type x);}
{Inserts \ccc{x} into the container and returns an iterator
pointing to the item inserted.}
\ccGlue
\ccMethod{void erase(iterator pos);}
{Removes from the container the item where \ccc{pos} points to.}
\ccGlue
\ccMethod{void swap(SimpleContainer other);}
{Swaps the two containers.}
\ccGlue
\ccMethod{void clear();}{Clears the contents of this container.}
\ccGlue
\ccMethod{size_type size();}{Returns the size of the container, i.e.,
the number of elements in the container.}
