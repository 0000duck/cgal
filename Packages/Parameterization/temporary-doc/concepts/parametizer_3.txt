// Concept of parameterization objects for a given type of mesh (MeshAdaptor_3)

class Parametizer_3 
{
// Public types
public:
				// The various errors detected by this package
				enum ErrorCode { 
					OK,
					ERROR_NON_CONVEX_BORDER,			// some parameterization algorithms require a convex border
					ERROR_BAD_MATRIX_CONDITIONING,		// result is mathematically unstable
					ERROR_NON_TRIANGULAR_MESH,			// the whole package is restricted to triangular meshes
					ERROR_NO_SURFACE_MESH,				// the whole package is restricted to surfaces
					ERROR_NO_1_TO_1_MAPPING,			// some parameterization algorithms do not ensure 1 to 1 mapping
					ERROR_EMPTY_MESH,					// cannot parameterize an empty mesh
					ERROR_NOT_ENOUGH_MEMORY,			// it's time to buy some RAM :-)
					ERROR_CANNOT_SOLVE_LINEAR_SYSTEM,	// cannot solve a sparse linear system (bug or bad matrix conditioning)
				};

// Public operations
public:
				// Compute a 1 to 1 mapping from a triangular 3D surface 'mesh' to a piece of the 2D space. The mapping is linear by pieces (linear in each triangle).
				// The result is the (u,v) pair image of each vertex of the 3D surface. 
				// 
				// Preconditions:
				// * 'mesh' must be a surface with 1 connected component and no hole
				// * 'mesh' must be a triangular mesh
				ErrorCode  parameterize (MeshAdaptor_3* mesh);
};
