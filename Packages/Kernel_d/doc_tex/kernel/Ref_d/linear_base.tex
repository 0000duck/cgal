\ccFunction{template <class ForwardIterator, class OutputIterator>
OutputIterator 
linear_base(ForwardIterator first, ForwardIterator last,
  OutputIterator result);}
{computes a basis of the linear space spanned by the vectors 
in \ccc{A = tuple [first,last)} and returns it via an iterator 
range starting in \ccc{result}. The returned iterator marks the 
end of the output.
\ccPrecond value type of \ccc{ForwardIterator} and \ccc{OutputIterator} 
is \ccc{Vector_d<R>} and $A$ contains vectors of the same dimension $d$.}

\end{ccRefFunction}
