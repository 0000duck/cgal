\begin{ccRefConcept}{SurfaceOracle}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries

\ccDefinition
  
The concept \ccRefName\ describes the required knowledge on the
surface to be meshed.

%\ccGeneralizes
%ThisConcept \\
%ThatConcept

\ccTypes

\ccNestedType{Point}{The type of points. This type is required to be
the type of points of the triangulation pointed to by the complex
storing the mesh.}
\ccNestedType{Segment}{The type of segments.}
\ccNestedType{Ray}{The type of rays.}
\ccNestedType{Line}{The type of lines.}



\ccCreation
\ccCreationVariable{surf}  %% choose variable name

\ccConstructor{SurfaceOracle();}{default constructor.}

\ccOperations


\ccMethod{Object intersect_segment_surface(Segment s);}
{ Returns an \ccc{Object}  which is an intersection \ccc{Point}
between the surface and the segment \ccc{s} or
a trivial \ccc{Object} if the intersection is empty.}
\ccGlue
\ccMethod{Object intersect_ray_surface(Ray r);}
{ Returns an \ccc{Object}  which is an intersection \ccc{Point}
between the surface and the ray \ccc{r} or
a trivial \ccc{Object} if the intersection is empty.}
\ccGlue
\ccMethod{Object intersect_line_surface(Line r);}
{ Returns an \ccc{Object}  which is an intersection \ccc{Point}
between the surface and the line \ccc{l} or
a trivial \ccc{Object} if the intersection is empty.}

\ccMethod{template <class OutputIteratorPoints>
OutputIteratorPoints
initial_points(OutputIteratorPoints pts);}
{Outpust a set of points on the surface.}
\ccGlue
\ccMethod{template <class OutputIteratorPoints>
OutputIteratorPoints
initial_points(OutputIteratorPoints pts, int n);}
{Outpust a set of \ccc{n} points on the surface.}


\ccHasModels

\ccc{Implicit_surface_oracle<Gt>}
