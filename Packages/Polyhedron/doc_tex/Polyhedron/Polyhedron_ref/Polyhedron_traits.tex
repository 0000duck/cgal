\begin{ccRefConcept}{Polyhedron_traits}
\label{pagePolyTraitsRef}  

A model for the \ccRefName\ concept must provide the following types
and operations. It defines the geometric types and primitive
operations used in the polyhedral surface data structure
\ccc{CGAL::Polyhedron_3<Traits>}.

\ccTypes

\ccNestedType{Point}{point type if points are supported.}
\ccGlue
\ccNestedType{Normal}{normal vector if normals are supported. Must be 
      assignment compatible to the normal vector type of \ccc{Plane} 
      if plane equations are supported.}
\ccGlue
\ccNestedType{Plane}{plane equation, if plane equations are
  supported.}

\ccOperations
\ccCreationVariable{traits}

\ccTagFullDeclarations

\ccMethod{void reverse_normal( Normal& n) const;}{
    reverses the orientation of the normal vector \ccc{n}. Only needed
    if normal vectors are and plane equations are not supported.}

\ccMethod{void reverse_plane( Plane& p) const;}{
    reverses the orientation of the plane equation \ccc{p}. Only
    needed if plane equations are supported.}

\ccTagDefaults

\ccHasModels

\ccc{CGAL::Polyhedron_default_traits_3}.

\ccSeeAlso

\ccc{CGAL::Polyhedron_3}.

\end{ccRefConcept}
