\ccRefPageBegin

\begin{ccRefConcept}{ArrangementOutputFormatter}

A model for the \ccRefName\ concept supports a set of functions that enable
writing an arrangement to an output stream using a specific format.

\ccTypes
%=======

\ccNestedType{Arrangement_2}{the type of arrangement to output.}

\ccTypedef{typedef typename Arrangement_2::Point_2 Point_2;}
    {the point type.}
\ccGlue
\ccTypedef{typedef typename Arrangement_2::X_monotone_curve_2 X_monotone_curve_2;}
    {the $x$-monotone curve type.}

\ccTypedef{typedef typename Arrangement_2::Size Size;}{}
\ccGlue
\ccTypedef{typedef typename Arrangement_2::Vertex_const_handle Vertex_const_handle;}{}
\ccGlue
\ccTypedef{typedef typename Arrangement_2::Halfedge_const_handle Halfedge_const_handle;}{}
\ccGlue
\ccTypedef{typedef typename Arrangement_2::Face_const_handle Face_const_handle;}{}

\ccCreation
\ccCreationVariable{outf}
%========================
    
\ccConstructor{Arr_out_formatter();}
   {default constructor.}

\ccConstructor{Arr_out_formatter (std::ostream& os);}
   {constructs a formatter that writes to \ccc{os}.}

\ccMethod{void set_out (std::ostream& os);}
   {directs \ccVar\ to write to \ccc{os}.}

\ccAccessFunctions
%=================

\ccMethod{std::ostream& out ();}
   {returns the stream that \ccVar\ writes to.
    \ccPrecond{\ccVar\ is directed to a valid output stream.}}

\ccHeading{Formatted Output Functions}
%=====================================

    {writes a message indicating the beginning of the arrangement.}

\ccMethod{void write_arr_end ();}
