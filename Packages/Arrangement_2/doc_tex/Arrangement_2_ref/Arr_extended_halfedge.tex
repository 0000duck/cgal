\begin{ccRefClass}{Arr_extended_halfedge<H,HData>}

\ccDefinition
%============
  The \ccRefName\ class-template extends the halfedge topological-features of
  the \dcel. It is parameterized by a halfedge-base type \ccc{H} and a data
  type \ccc{HData} used to extend the halfedge base type.

\ccInclude{CGAL/Arr_extended_halfedge.h}

\ccIsModel
%============
  \ccc{ArrangementDcelHalfedge}

\ccInheritsFrom
%============
  \ccc{H}

\ccCreation
\ccCreationVariable{he}
%==========
  \ccMethod{void assign (const Self & other);}
  {assigns \ccVar{} with the contents of the \ccc{other} vertex.}

\ccAccessFunctions
%=================
  \ccMethod{HData & data ();}
  {obtains the auxiliary data.}

\ccModifiers
%============

  {sets the auxiliary data.}
