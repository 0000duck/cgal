\begin{ccRefConcept}{SkinSurfaceTraits_3}
\label{pageSkinSurfaceTraits}  

Required types and member functions for the \ccRefName\ concept. This
geometric traits concept is used for the construction of a polyhedral
mesh approximating a skin surface \ccc{CGAL::skin_surface_3<Traits>}.
It defines three \cgal\ kernels and the appropriate converters.

\ccRefines

%\ccc{CopyConstructable}, \ccc{Assignable}.

\ccTypes

\ccNestedType{Regular_kernel}{This is the \ccc{kernel} used in the
  regular triangulation.}

\ccNestedType{Triangulated_mixed_complex_kernel}{This is the
  \ccc{kernel} used in the \ccc{Triangulated_mixed_complex}.}

\ccNestedType{Polyhedron_kernel}{This is the \ccc{kernel} used for the
  geometric objects defined by the mesh.}

\ccNestedType{R2T_converter}{Converter to convert geometric objects
  defined within \ccc{Regular_kernel} to objects defined within
  \ccc{Triangulated_mixed_complex_kernel}.}

\ccNestedType{R2P_converter}{Converter to convert geometric objects
  defined within \ccc{Regular_kernel} to objects defined within
  \ccc{Polyhedron_kernel}.}

\ccNestedType{T2P_converter}{Converter to convert geometric objects
  defined within \ccc{Triangulated_mixed_complex_kernel} to objects
  defined within \ccc{Polyhedron_kernel}.}

\ccNestedType{T2P_converter}{Converter to convert geometric objects
  defined within \ccc{Polyhedron_kernel} to objects
  defined within \ccc{Triangulated_mixed_complex_kernel}.}
