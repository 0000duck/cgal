\begin{ccRefConcept}{MeshTraits_2}

\ccDefinition

In addition of the requirements of the concept
\ccc{DelaunayTriangulationTraits_2}, the concept \ccRefName\ has to
provide a numeric field type \ccc{FT}, an enumeration \ccc{Angle}, a
type \ccc{Vector_2} and several constructors on \ccc{Vector_2} and
\ccc{Point_2}, a predicate on angles \ccc{Angle_2}, and two special
predicates \ccc{Compute_squared_minimum_sine}, and \ccc{Is_bad}.

\ccRefines

\ccc{DelaunayTriangulationTraits_2}

\ccTypes

\ccNestedType{FT}{The field type.}  \ccNestedType{Vector_2}{The vector
  type.}
\ccNestedType{Construct_vector_2}{Constructor object. Must
  provide the operator \ccc{Vector_2 operator()(Point a, Point b)}
  that computes the vector $b-a$.}
\ccNestedType{Construct_scaled_vector_2}{Constructor object. Must
  provide the operator \ccc{Vector_2 operator()(Vector_2 v, FT scale)}
  that computes the vector $scale \cdot \mathbf{v}$.}
\ccNestedType{Construct_translated_point_2}{Construct object. Must
  provide the operator \ccc{Point_2 operator()(Point_2 p, Vector_2 v)}
  that computes the point $p + \mathbf{v}$.}
\ccNestedType{Construct_midpoint_2}{Constructor object. Must provide
  the operator \ccc{Point_2 operator()(Point_2 a, Point_2 b)} that
  computes the middle point of the segment $ab$.}
\ccEnum{enum Angle { OBTUSE, RIGHT, ACUTE};}{Enumeration needed by the
  following type}
\ccNestedType{Angle_2}{Predicate object. Must provide the operator
  \ccc{Angle operator()(Point_2 p, Point_2 q, Point_2 r)} that OBTUSE,
  RIGHT or ACUTE depending on the angle formed by the three points p,
  q, r (q being the vertex of the angle).}
\ccNestedType{Compute_squared_minimum_sine_2}{Constructor object. Must 
  provide the operator \ccc{double operator()(Point_2 a, Point_2 b,
    Point_2 c)} that computes the squared minimum sine of the triangle 
  formed by the three points a, b, c.}
\ccNestedType{Is_bad}{Predicate object. Must provide the operator
  \ccc{bool operator()(Point_2 a, Point_2 b, Point_2 c)} with tells if 
  the triangle formed by by the three points a, b, c satisfies the
  wanted criteria for mesh triangles.}
