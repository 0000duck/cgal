\begin{ccRefFunction}{make_conforming_Delaunay_2}


\ccFunction{ template<class Tr> void make_conforming_Delaunay_2 (Tr &t); } 
{ Conforms the constrained edges of \ccc{t} regarding the Delaunay
  criteria. After a call to this functions, the triangulation is a
  Delaunay triangulation.
  \ccPrecond \ccc{Tr} has to be a \cgal\ constrained Delaunay
  triangulation (CDT) and its geometric traits class must be a model
  of \ccc{ConformingDelaunayTriangulationTraits_2}.}

\end{ccRefFunction}
