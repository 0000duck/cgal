\begin{ccRefClass}{Mesh_size_traits_2<K>}

\ccCreationVariable{traits}
\ccDefinition
  
The class \ccRefName\ is a model for the \ccc{MeshTraits_2},
\ccc{TriangulationTraits_2} and \ccc{DelaunayTriangulationTraits_2}
concepts. It uses types defined in \cgal\ kernel as geometric
primitives types. The shape constraint on triangle is given by a bound
$B$ such that good triangles verify $\frac{r}{l} \le B$ where $l$ is
the minimum length and $r$ is the circumradius of the triangle. This
traits class defines also a size constraint: all segments of all
triangles must be smaller than a bound $S$.

\ccInclude{CGAL/Mesh_size_traits_2.h}

\ccIsModel

\ccc{MeshTraits_2} \\
\ccc{TriangulationTraits_2} \\
\ccc{DelaunayTriangulationTraits_2}

\ccCreation
\ccConstructor{Mesh_default_traits_2()}{Default
  constructor. $B=\sqrt{2}$. No bound on size}
\ccConstructor{Mesh_default_traits_2(double b = 0.125, double s =
  0)}{Construct a traits class with bound $B=\sqrt{\frac{1}{4
      b}}$. If s $\neq 0$, the size bound is s. If s $= 0$, there is
  no bound on size. }

\end{ccRefClass}
