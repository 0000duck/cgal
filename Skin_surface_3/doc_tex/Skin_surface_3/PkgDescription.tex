\begin{ccPkgDescription}{3D Skin Surface Meshing}
  \ccPkgSummary{ This package allows to build a triangular mesh of a
    skin surface, defined by a set of weighted points. Several
    algorithms are provided for the construction of the mesh.}
  
  \ccPkgDependsOn{Triangulation_3 and Polyhedron}
  \ccPkgMaturity{Not introduced in CGAL yet}
