\begin{ccPkgDescription}{3D Skin Surface Meshing \label{Pkg:SkinSurface3}}
  \ccPkgSummary{ This package allows to build a triangular mesh of a
    skin surface, defined by a set of weighted points. Several
    algorithms are provided for the construction of the mesh.}

  \ccPkgIntroducedInCGAL{3.3}
  \ccPkgDependsOn{\ccRef[3D Triangulation]{Pkg:Triangulation_3} and \ccRef[3D Polyhedral Surface]{Pkg:Polyhedron}}

\end{ccPkgDescription}
