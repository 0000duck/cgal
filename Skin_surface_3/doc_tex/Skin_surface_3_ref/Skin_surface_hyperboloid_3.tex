\begin{ccRefClass}{Skin_surface_hyperboloid_3<Kernel>}
\label{refSkinsurfacehyperboloid}

\ccDefinition
%
The class \ccc{Skin_surface_hyperboloid_3} defines a hyperbolic
surface that is used in the construction of the skin surface mesh.


\ccInclude{CGAL/Skin_surface_quadratic_surface_3.h}

\ccParameters 
%
It is parameterized by a kernel that must provide points, vectors,
segments and weighted points.

\ccInheritsFrom{MarchingTetrahedraTraits_3}

\ccTypes
\ccThree{typedef typename Triangulation_3::Cell_handle}{xxx}{ }
\ccThreeToTwo

\ccTypedef{typedef Kernel::Point_3}{Point;}
\ccGlue
\ccTypedef{typedef Kernel::Vector_3}{Vector;}
\ccGlue
\ccTypedef{typedef Kernel::Segment_3}{Segment;}
\ccGlue
\ccTypedef{typedef Kernel::RT_3}{RT;}

\ccCreation
\ccCreationVariable{surf}  %% choose variable name

\ccConstructor{ Skin_surface_hyperboloid_3(Weighted_point wp, Vector
  t, RT s, int orient)}{%
  Constructs a hyperboloid defined by an edge or facet of the regular
  triangulation that is part of the skin surface. The weighted point
  \cc{wp} is the center of the Delaunay simplex, the vector \cc{t} is
  the symmetry axis of the hyperboloid and $s$ is the shrink factor.
  The orientation is -1 or 1 and used to exchange the inside and
  outside, which is the difference between a hyperboloid defined by a
  Delaunay edge and a Delaunay facet.}

\ccOperations

\ccMethod{RT value(const Point &p) const;}{Returns the function value
  of the hyperboloid.}
\ccGlue
\ccMethod{Vector gradient(const Point &p); }{Returns the gradient in
  \ccc{p}. }
