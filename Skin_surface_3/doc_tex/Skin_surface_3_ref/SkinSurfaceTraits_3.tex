\begin{ccRefConcept}{SkinSurfaceTraits_3}
  Required types and member functions for the \ccRefName\ concept.
  This geometric traits concept is used for the construction of a
  polyhedral mesh approximating a skin surface
  \ccc{CGAL::skin_surface_3<Traits>}.

  \ccRefines
  \ccc{RegularTriangulationTraits_3} 

  % \ccc{CopyConstructable}, \ccc{Assignable}.

  \ccThree{Skin_surface_traits_3}{Triang}{}
  \ccThreeToTwo
  \ccTypes

  \ccNestedType{RT}{The number type.}

  \ccNestedType{Construct_anchor_point_3}{A predicate object which
    provides the following function operator:\\
    \ccc{Point operator()(const Point &center_del, const
      Point &center_vor);}\\
    Constructs the anchor point in between the Delaunay and Voronoi
    centers, i.e., the point $(1-s)\cdot$\ccc{center_del} +
    $s\cdot$\ccc{center_vor}, where $s$ is the shrink factor.}

\ccCreation
\ccCreationVariable{traits}

\ccConstructor{SkinSurfaceTraits_3(RT s=.5);}{
    constructor that takes the shrink factor as argument.}

\ccHasModels

\ccThreeToTwo

  const;}{returns a \ccc{Construct_anchor_point_3} object.}
