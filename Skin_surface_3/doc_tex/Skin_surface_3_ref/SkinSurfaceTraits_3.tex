\begin{ccRefConcept}{SkinSurfaceTraits_3}
  NGHK: TODO

  NGHK: USES SkinSurfaceTraits::Bare_point::R in Weighted_converter! CHECK


  Required types and member functions for the \ccRefName\ concept.
  This geometric traits concept is used for the construction of a
  polyhedral mesh approximating a skin surface
  \ccc{CGAL::skin_surface_3<Traits>}.  It defines three \cgal\ kernels
  and appropriate converters.

%\ccRefines
%\ccc{CopyConstructable}, \ccc{Assignable}.

\ccThree{Skin_surface_traits_3xxx}{Triangulated}{}
\ccThreeToTwo
\ccTypes

\ccNestedType{Regular_triangulation_traits}{The traits class is used for the
  regular triangulation. It is a model of the
  \ccc{RegularTriangulationTraits_3} concept.}

\ccNestedType{Triangulated_mixed_complex_traits}{The traits
  class used in the \ccc{Triangulated_mixed_complex}. It is a model of
  the \ccc{TriangulationTraits_3} concept and has to be able to store
  the constructed vertices exactly.}

\ccNestedType{Polyhedron_traits}{The traits class used for the
  geometric objects defined by the mesh.}

\ccNestedType{Regular_RT}{The ring type of the
  \ccc{Regular_triangulation_traits}.}

\ccNestedType{R2T_converter}{Converter for geometric objects
  defined within \ccc{Regular_triangulation_traits} to objects defined within
  \ccc{Triangulated_mixed_complex_traits}.}

\ccNestedType{R2P_converter}{Converter for geometric objects
  defined within \ccc{Regular_triangulation_traits} to objects defined within
  \ccc{Polyhedron_traits}.}

\ccNestedType{T2P_converter}{Converter for geometric objects
  defined within \ccc{Triangulated_mixed_complex_traits} to objects
  defined within \ccc{Polyhedron_traits}.}

\ccNestedType{P2T_converter}{Converter for geometric objects
  defined within \ccc{Polyhedron_traits} to objects
  defined within \ccc{Triangulated_mixed_complex_traits}.}
