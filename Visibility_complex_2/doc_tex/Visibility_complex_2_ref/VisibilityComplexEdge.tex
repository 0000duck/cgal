\begin{ccRefConcept}{VC2Edge}

\ccDefinition

The concept \ccRefName{} represents the edge class in the visibility
complex.  It defines the minimal requirements for the \ccc{Edge} type in
the \ccc{Visibility_complex_2<VC2GeomTraits,VC2Items>} class. It is also
required in the \ccc{Edge_wrapper<Vc>} member class template of an items
class, see the \ccc{VC2Items} concept.

\ccTypes
\ccThree{typedef Edge::Vertex_handle}{Vertex_handle;}{}
\ccThreeToTwo

\ccNestedType{Gt} {Geometric traits class.}
\ccNestedType{Disk}{Same as \ccc{Gt::Disk}.}
\ccGlue
\ccNestedType{Vertex}{Model of \ccc{VC2Vertex}.}
\ccGlue

the containing class.}

\ccGlue

\ccGlue

\ccGlue
\ccNestedType{Face_const_handle}{Const handle to faces.}
\ccGlue

\ccThreeToTwo

\ccCreation
\ccCreationVariable{e}

\ccConstructor{Edge();}{Default constructor.}
\ccConstructor{Edge(bool s, Disk_handle p);}
{Creates an edge with positive sign if $s$ is true and negative otherwise.
The underlying arc is the complete boundary of the disk $p$.}
\ccConstructor{Edge(Vertex_handle v, Vertex_handle w , Disk_handle p);}
{Creates an edge on disk $p$ whose source is $v$ and sink $w$. The underlying
arc is defined by the extremities of $v$ and $w$ on $p$. }

\ccAccessFunctions

\ccTagFullDeclarations
\ccMethod{Face_handle dl();}{Returns the lower left face.}
\ccGlue
\ccMethod{Face_handle dr();}{Returns the lower right face. Same as
                             \ccc{dl()} for a positive edge.}
\ccGlue
\ccMethod{Face_handle ul();}{Returns the upper left face.}
\ccGlue
\ccMethod{Face_handle ur();}{Returns the upper right face. Same as
                             \ccc{ul()} for a negative edge.}
\ccGlue
\ccMethod{Vertex_handle sup();}
    {Returns the sink vertex of \ccVar.}
\ccGlue
\ccMethod{Vertex_handle inf();}
    {Returns the source vertex of \ccVar.}

All the latter methods have const counterparts returning const handles.

\ccMethod{bool sign(); const}{Returns the sign of \ccVar.}
\ccGlue
\ccMethod{bool is_regular(); const}{Returns true iff \ccVar is supported by
an object, as opposed to being a cusp-edge of a constraint.}
\ccGlue
\ccMethod{Disk_handle object(); const}{Returns the disk supporting \ccVar. If
\ccVar is a cusp-edge of a constraint, returns an invalid handle.}

\ccOperations
\ccMethod{void set_adjacent_faces(Face_handle f0,Face_handle f1,Face_handle f2);} {set
    \ccc{f0,f1,f2} as the adjacent faces of \ccVar. For a positive edge,
    \ccc{f0 = dl() = dr()}, \ccc{f1 = ur()} and \ccc{f2 = ul()}. For a
    negative edge \ccc{f0 = dl()}, \ccc{f1 = dr()} and \ccc{f2 = ul() =
    ur()}.}
\ccGlue
\ccMethod{void set_inf(Vertex_handle v);}
    {sets the source vertex of \ccVar\  to $v$ and sets the \ccc{ccw_edge}
    of $v$ to \ccVar.}
\ccGlue
\ccMethod{void set_sup(Vertex_handle v);}
    {sets the sink vertex of \ccVar\  to $v$ and sets the \ccc{cw_edge}
    of $v$ to \ccVar.}
