\begin{ccRefConcept}{VC2GeomTraits}

\ccDefinition

The concept \ccRefName{} describes the set of requirements to be fulfilled
by any class used to instantiate the first template parameter of the
classes \ccc{Visibility_complex_2} and \ccc{Compute_free_bitangents_2}.
This concept provides the types of the geometric primitives used in the
visibility complex and some function object types for the required
predicates on those primitives.

\ccTypes

\ccNestedType{R}{Underlying kernel.}
\ccGlue
\ccNestedType{Point_2}{Point type.}
\ccGlue
\ccNestedType{Disk}{Disk type.}
\ccGlue
\ccNestedType{Bitangent_2}{Model of the \ccc{VC2Bitangent} concept.}
\ccGlue
\ccNestedType{Arc_2}{Model of the \ccc{VC2Arc} concept.}
\ccGlue
\ccNestedType{Distance_NT}{(optional, used only for shortest path
                           computations) A number type supporting sqrt.}

\ccOperations
\ccThree{VisibilityComplexTraits::Is_upward_directed }{}{}
\ccThreeToTwo

\ccNestedType{Orientation_object}{Must provide the operator
\ccc{Orientation operator()(const Bitangent_2& a, const Bitangent_2& b)},
which returns \ccc{LEFTURN, RIGHTTURN, COLLINEAR} depending on the
orientation of the angle between \ccc{a} and \ccc{b}.}
\ccGlue
\ccNestedType{Supports_chi3}{Defined as \ccc{Tag_true} or \ccc{Tag_false},
tells whether \ccc{Orientation_infinite} does work.}
\ccGlue
\ccNestedType{Orientation_infinite}{Must provide the operator
\ccc{Orientation operator()(const Bitangent_2& a, const Bitangent_2& b)},
which returns \ccc{LEFTURN, RIGHTTURN, COLLINEAR} depending on whether the
source tangency point of \ccc{b} is above, upon or below the supporting
oriented line of \ccc{a}. Behaviour is undefined if
\ccc{Orientation_infinite} is defined as \ccc{Tag_false}.}
\ccGlue
\ccNestedType{Is_upward_directed}{Must provide the operator \ccc{bool
operator()(const Bitangent_2& d)}, which returns true iff. the angle of the
bitangent with the x-axis belongs to $[0,\pi)$. }
\ccGlue
\ccNestedType{Compare_extreme_yx}{
Provides 
\ccc{Comparison_result operator()(bool x,const Disk& a,bool y,const Disk& b)},
that y-compares the y-coordinate of the y-extreme points of a and b. The
booleans select between the lower (true) and upper (false) extreme points.}
\ccGlue
\ccNestedType{Equal_as_segments}{Must provide the operator \ccc{bool
operator()(const Bitangent_2& a, const Bitangent_2& b)}, which returns true
iff. the underlying segments of $a$ and $b$ are equal.}

The following functors are optional, useful only for shortest path
computations.

\ccNestedType{Make_convex_from_point}{Functor to convert a point into a disk.}
\ccNestedType{Length}{Functor to compute the length of a bitangent or of an
arc.}
